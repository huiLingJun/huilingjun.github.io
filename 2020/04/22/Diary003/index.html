二、 三级多选框 的 两种实现方式

Way1–遍历子选框

一、遍历子选框
    <div v-for="(item, index) in contentList" :key="index" class="box-card">
        <el-checkbox
          :indeterminate="item.isIndeterminate"
          v-model="item.checkAll"
          @change="handleCheckChange($event, item)"
          class="checkTitle"
        >{{item.name}}</el-checkbox>
        <div style="margin: 15px 0;"></div>
        <el-checkbox-group v-model="item.checkedItems" @change="handleCheckedItemsChange($event, item)">
            <el-checkbox
              v-for="(citem, cindex) in item.functionList"
              :key="cindex"
              :label="citem"
              class="chart"
            >{{citem.name}}</el-checkbox>
        </el-checkbox-group>
    </div>
    <div class="foot">
        <el-checkbox
          :indeterminate="indeterminate"
          v-model="ischeckAll"
          @change="handleCheckAllChange"
        >{{fn["选择全部"][value]}}</el-checkbox>
    </div>

    methods: {
        getData() {
          this.$api.memberPower(this.$store.state.userId)
            .then(res => {
              if (res.data.code == 200) {
                let resData = res.data.result;
                resData.forEach(item => { item.checkAll = false; item.isIndeterminate = false; item.checkedItems = [];
                  item.functionList.forEach(ele => { 
                    this.checkAllListIds.push(ele.id)
                    if(ele.resourceType == 1) {     //默认选中
                      item.checkedItems.push(ele)
                    }
                  })
                  this.contentList.push(item)
                })
              } else {
                this.$message({ message: res.data.message, type: "error" })
              }
            })
            .catch(err => {
              console.error(err);
            });
        },
        handleCheckChange(value, item) {
          item.checkedItems = value ? item.functionList : [];
          item.isIndeterminate = false;
        },
        handleCheckedItemsChange(value, item) {
          let checkedCount = value.length;
          item.checkAll = checkedCount === item.functionList.length;
          item.isIndeterminate = checkedCount > 0 && checkedCount < item.functionList.length;
        },
        handleCheckAllChange(e) {
          this.ischeckAll = e;
          this.indeterminate = false;
          for(var i=0; i<this.contentList.length; i++){ 
            this.contentList[i].checkAll = e;
            this.contentList[i].isIndeterminate = false; //去掉二级不确定状态
            this.contentList[i].checkedItems = e ? this.contentList[i].functionList : [];
          }
        },
    }

Way2–单独子选框

二、单独子选框
    <div v-for="(item, index) in contentList" :key="index" class="box-card">
      <el-checkbox
        :indeterminate="item.isIndeterminate"
        v-model="item.checkAll"
        @change="handleCheckChange($event, item)"
        class="checkTitle"
      >{{item.name}}</el-checkbox>
      <div class="elcheck">
        <div v-for="(citem, cindex) in item.functionList" :key="cindex" class="chart">
          <el-checkbox 
            v-model="citem.mychecked"
            @change="handleCheckedItemsChange(index)"
            :title="citem.name"
            :label="citem.id"
          >{{citem.name}}</el-checkbox>
        </div>
      </div>
    </div>
    <div class="foot">
      <el-checkbox
        :indeterminate="indeterminate"
        v-model="ischeckAll"
        @change="handleCheckAllChange"
      >{{fn["选择全部"][value]}}</el-checkbox>
    </div>
    <div style="text-align: center;">
      <el-button type="success" size="medium" @click="handleKeep()">{{fn["保存"][value]}}</el-button>
    </div>

    methods: {
        getData() {
          this.$api.departPower(this.$store.state.departId)
            .then(res => {
              if (res.data.code == 200) {
                let resData = res.data.result;
                resData.forEach(item => { item.checkAll = false; item.isIndeterminate = false; 
                  item.functionList.forEach(ele => { 
                    ele.mychecked = false;
                    this.checkAllListIds.push(ele.id)
                    if(ele.resourceType == 1) {
                      ele.mychecked = true;
                    }
                  })
                  this.contentList.push(item)
                })
              } else {
                this.$message({ message: res.data.message, type: "error" })
              }
            })
            .catch(err => {
              console.error(err);
              this.$message.error(err)
            });
        },
        handleCheckChange(value, item) {
          if(value == true) { 
            item.checkAll = true; 
            item.isIndeterminate = false;
            item.functionList.forEach(ele => { ele.mychecked = true })
          } else { 
             item.checkAll = false; 
             item.functionList.forEach(ele => { ele.mychecked = false })
          }  
        },
        handleCheckedItemsChange(index) {
          let funs = this.contentList[index].functionList;
          var tickCount = 0, unTickCount = 0;
          for(var i=0; i<funs.length; i++) {
            if (funs[i].mychecked == true) {
              tickCount++;
            }
            if (funs[i].mychecked == false) {
              unTickCount++;
            }
          }
          if (tickCount == funs.length) {            
            this.contentList[index].checkAll = true;
            this.contentList[index].isIndeterminate = false;
          } else {
            this.contentList[index].checkAll = false;   
            this.contentList[index].isIndeterminate = true;
          }
        },
        handleCheckAllChange(e) {
          this.ischeckAll = e;
          this.indeterminate = false;
          for(var i=0; i<this.contentList.length; i++){ 
            this.contentList[i].checkAll = e;
            this.contentList[i].isIndeterminate = false; 
            let funs = this.contentList[i].functionList;
            for(var j=0; j<funs.length; j++) { funs[j].mychecked = e }
          }
        },
        handleKeep() {
          let obj = {};
          obj.id = this.$store.state.departId;
          if(this.ischeckAll == true) {
            obj.functionIds = this.checkAllListIds;
          } else {
            this.checkListIds = [];
            this.contentList.forEach(item => {
              item.functionList.forEach(citem => {
                if(citem.mychecked == true) {
                  this.checkListIds.push(citem.id)
                }
              })
            })
            obj.functionIds = this.checkListIds;
          }
          this.$api.departSubmit(obj).then(res => {
            if(res.data.code == 200) {
              this.$message.success(this.fn["保存成功"][this.value])
            } else {
              this.$message.error(res.data.message)
            }
          }).catch(err => { console.error(err) })
        }
    }

三、 使用客户的语言包(后台返回数据)做 语言切换

登录页：

首先，在登录时拿到后台返回的语言包数据：在登录页面操作：
<script>
    import store from "../store/store";
    if(localStorage["fn"]) { var getFn = JSON.parse(localStorage["fn"]); }     //***
    export default {
        name: "login",
        store,
        data() {
          return {
            fn: getFn? getFn: {},   //***
            value: localStorage["value"],   //***
            rules: {
              user: [
                {
                  required: true,
                  message: getFn? getFn["请输入账号"][this.value] : "请输入账号",
                  trigger: "blur"
                }
              ],
              password: [
                {
                  required: true,
                  message: getFn? getFn["请输入密码"][this.value] : "请输入密码",
                  trigger: "blur"
                }
              ]
            }
          };
        },

        beforeCreate() {    //***
            if (typeof localStorage["value"] == "undefined") {
              localStorage.setItem("value", "chinese");
            }
            if (typeof localStorage["fn"] == "undefined") {
              this.$api.getVersion().then(res => {
                let version = res.data.result;
                let obj = {};
                obj.version = version;
                this.$api.getLanguage(obj).then(res => {
                  let resData = res.data.result;
                  let fn = {};
                  for (var i = 0; i < resData.length; i++) {
                    let name = resData[i].chinese;
                    fn[name] = resData[i];
                  }
                  localStorage["fn"] = JSON.stringify(fn);
                  this.fn = JSON.parse(localStorage["fn"])
                });
              });
            }
        },
    }
</script>

语言设置页：

在语言设置页面：
<template>
    <div class="language-box">
      <span>{{fn["系统语言"][value]}}: </span>
      <el-select v-model="language" :placeholder='fn["请选择"][value]'>
        <el-option v-for="item in options" :key="item.label" :label="item.label" :value="item.value"></el-option>
      </el-select>
      <div style=" margin-top:50px ;">
        <el-button type="success" @click="submitForm()">{{fn["提交"][value]}}</el-button>
      </div>
    </div>
</template>
<script>
    var getVersion = localStorage.getItem('version')      //***
    var getFn = JSON.parse(localStorage['fn'])      //***

    export default {
      inject: ["reload"],
      data() {
        return {
          options: [
            { value: "chinese" ,label:'中文'},
            { value: "english" ,label:'英文'},
            // { value: "taiwen" ,label:'泰文'},
            { value: "zawgyi" ,label:'缅文（zawgyi）'},
            { value: "mm3" ,label:'缅文（mm3）'},
          ],
          language: "",
          version: 1,
          fn: getFn,        //***
          value: localStorage['value'],        //***
        };
      },
      mounted() {
        this.fetchVersion();
      },
      methods: {
        fetchVersion() {
          this.$api.getVersion().then(res => {
            if (res.data.code == 200) {
              this.version = res.data.result;
              if(getVersion = null || this.version != getVersion) {
                this.resetForm()
              }
              localStorage.setItem('version', this.version)
            }
          });
        },
        resetForm() {
          let data = {};
          data.version = this.version;
          this.$api.getLanguage(data).then(res => {
            let resData = res.data.result;
            let fn = {};
            for(var i=0; i<resData.length; i++) {
              let name = resData[i].chinese;
              fn[name] = resData[i];
            }
            localStorage['fn'] = JSON.stringify(fn)
          });
        },
        submitForm() {
          if(this.language == "chinese") {
            this.value = "chinese";
            localStorage.setItem('value', this.value)
          } else if (this.language == "english") {
            this.value = "english";
            localStorage.setItem('value', this.value)
          } else if (this.language == "zawgyi") {
            this.value = "zawgyi";
            localStorage.setItem('value', this.value)
          } else if (this.language == "mm3") {
            this.value = "mm3";
            localStorage.setItem('value', this.value)
          }
          this.$router.go(0)
        }
      }
    };
</script>

四、 element组件 el-upload踩坑

一般情况下，el-upload上传组件的上传地址一般是放在action，但是由于action是指定一条路径写死的，有时候我们的需求是要请求接口上传，而action里面不能够做略复杂的请求，因而出现很多坑，其实也可以自己写一个上传方法，但是本宝宝懒得写，故使用如下手动上传的办法：
<el-form-item label="更新版本号" prop="versionInt" class="elUpload">
      <el-input v-model="versionForm.versionInt" disabled></el-input>
      <el-upload
        ref="upload"
        class="upload-demo"
        action="https://jsonplaceholder.typicode.com/posts/"    //这里使用一个假地址欺骗一下这个组件，因为action是必填的，避免控制台一直报错比较难看
        :headers="myheaders"
        :on-preview="handlePreview"
        :on-remove="handleRemove"
        :limit="1"
        :file-list="fileList"
        :auto-upload="false"
        :on-change="handleChange"
      >
        <el-button size="small" type="primary" slot="trigger">选取文件</el-button>
        <el-button size="medium" type="success" style="margin-left:10px" @click="submitUpload">上传程序</el-button>     //***使用手动上传方式
         <div slot="tip" class="el-upload__tip">* 文件不能超过10M</div>
      </el-upload>
</el-form-item>

data() {
    return {
        fileList: [],
        myheaders: { "fsd-token": sessionStorage.getItem("token") },
        row: null,
        appName: null,
        fileName: null,
        fdd: {}
    }
},

methods: {
    //上传状态改变的方法 (对应上面的:on-change)
    handleChange(file,fileList) {
      if (file.size >= 10 * 1024 * 1024) {
        this.$message.error("文件大小不能超过10M");
        return false;
      } else {
        this.fdd = new FormData();
        this.fdd.append("upLoadFile", file.raw); //传文件
      }
      this.fileList = fileList
    },
    submitUpload() {
      this.$api.uploadApp(this.appName, this.fdd).then(res => {
        if(res.data.code == 200) {
          this.$message.success("上传成功!");
          this.versionForm.versionInt = res.data.result.versionInt;
          this.versionForm.versionStr = res.data.result.versionStr;
          this.fileName = res.data.result.fileName;
        } else {
          this.$message.error(res.data.message);
          this.$refs.upload.clearFiles();
        }
      });
    },
    handleRemove(file, fileList) {
      console.log(file, fileList);
    },
    handlePreview(file) {
      console.log(file);
    },

    //上传完最后的保存方法可以这么写：
    saveData() {
        let data = new FormData();
        if (this.fileList) {
          data.append("fileName", this.fileName);
        }
        data.append("appName", this.appName);
        if (this.versionForm.versionName) {
          data.append("versionName", this.versionForm.versionName);
        }
        if (this.versionForm.Remark) {
          data.append("updateMsg", this.versionForm.Remark);
        }
        if (this.fileList) {
          data.append("upLoadFile", this.fileList[0].raw)
        }
        if (this.fileList < 0 && this.versionForm === null) {
          this.$message.error("请输入数据")
          return;
        }
        this.$api.updateApp(data)
          .then(res => {
            if (res.data.code == 200) {
              this.$message.success("提交成功!");
              this.dialogVersions = false;
              this.versionForm = {};
              this.getVersion();
            } else {
              this.$message.error(res.data.message)
            }
          })
          .catch(function(err) { console.error(err) });
    }
}

五、 下载文件方法(blob)

loadProgram(index, row) {
  axios({
    method: "get",
    url: "appVersion/download?appName=" + row.appName,
    responseType: "blob"
  }).then(res => {
    let url = window.URL.createObjectURL(new Blob([res.data]));
    let link = document.createElement("a");
    link.style.display = "none";
    link.href = url;
    link.setAttribute("download", row.appName+".apk");
    document.body.appendChild(link);
    link.click();
    this.$message.success("下载成功!")
  }).catch(err => {
    console.error(err)
    this.$message.error(err)
  })
},
